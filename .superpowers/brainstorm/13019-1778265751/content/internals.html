Monitor Interface

type Monitor interface {
    Name()  string
    Check(ctx context.Context) (CheckResult, error)
}

type CheckResult struct {
    MonitorName string
    Status      CheckStatus  // OK | ERROR | UNKNOWN
    Value       any          // raw value from check
    Message     string       // human readable detail
    Timestamp   time.Time
}

// Each monitor runs in its own goroutine:
func runMonitor(ctx context.Context, m Monitor,
    interval time.Duration,
    results chan<- CheckResult) {

    ticker := time.NewTicker(interval)
    defer ticker.Stop()
    for {
        select {
        case <-ticker.C:
            r, err := m.Check(ctx)
            if err != nil {
                r = CheckResult{..., Status: UNKNOWN}
            }
            results <- r
        case <-ctx.Done():
            return
        }
    }
}

Action Interface

type Action interface {
    Name() string
    Fire(ctx context.Context, event AlarmEvent) error
}

type AlarmEvent struct {
    MonitorName string
    Transition  Transition   // FIRING | RESOLVED
    Message     string
    Value       any
    FiredAt     time.Time
}

// HTTPAction implements Action for ntfy.sh / webhooks
type HTTPAction struct {
    name   string
    cfg    HTTPActionConfig
    client *http.Client
}

Alarm State & Evaluator

type AlarmState struct {
    MonitorName   string
    Status        AlarmStatus  // OK|FIRING|UNKNOWN
    LastCheck     time.Time
    FiredAt       *time.Time   // nil if OK
    PendingSince  *time.Time   // for "for:" duration
    LastValue     any
    Message       string
}

type HistoryEvent struct {
    MonitorName  string
    Transition   Transition
    At           time.Time
    Message      string
}

// Evaluator logic (per CheckResult received):
// 1. Evaluate threshold against result.Value
// 2. If threshold met and no PendingSince → set PendingSince=now
// 3. If threshold met and now-PendingSince >= for → FIRING
// 4. If threshold not met → clear PendingSince, set OK
// 5. On state transition → append HistoryEvent, call Dispatcher
// 6. Mark state dirty for next StateWriter flush

State Writer (ConfigMap)

// Stored as JSON in klyra-state ConfigMap
type PersistedState struct {
    UpdatedAt time.Time
    Alarms    map[string]AlarmState
    History   []HistoryEvent   // trimmed to 24h
    Silences  []Silence
}

// Writer ticks every 10s, only writes if dirty
// Before write: drops HistoryEvents older than 24h
// Non-leader replicas read ConfigMap every 30s for UI
// ~50 monitors × ~200 bytes + history ≈ well under 1MB
